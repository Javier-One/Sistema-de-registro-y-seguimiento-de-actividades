Configurar sincronización con Google Sheets
Apps Script corregido — sigue estos pasos exactamente.
1
Abrir el Google Sheets convertido (nativo, no .xlsx)
Debe tener el menú "Extensiones" visible en la barra superior.
2
Extensiones → Apps Script → borrar todo el código existente → pegar este:
const SHEET_NAME = 'Actividades'; function doGet(e) { return handleReq(e); } function doPost(e) { return handleReq(e); } function handleReq(e) { const lock = LockService.getScriptLock(); lock.tryLock(10000); try { let data; if (e.postData && e.postData.contents) { data = JSON.parse(e.postData.contents); } else if (e.parameter && e.parameter.data) { data = JSON.parse(e.parameter.data); } else { return ok({status:'activo', mensaje:'Apps Script funcionando correctamente'}); } if (data.action === 'addActivity') return addActivity(data.data); return ok({status:'error', mensaje:'Accion no reconocida'}); } catch(err) { return ok({status:'error', mensaje: err.toString()}); } finally { lock.releaseLock(); } } function addActivity(d) { const ss = SpreadsheetApp.getActiveSpreadsheet(); const sheet = ss.getSheetByName(SHEET_NAME); const lastRow = Math.max(sheet.getLastRow(), 2); sheet.appendRow([ lastRow - 1, d.fecha, d.persona, d.cargo, d.actividad, d.tipo, d.horaInicio, d.horaFin, d.horas, d.tipo, d.observaciones || '' ]); return ok({status:'ok'}); } function ok(data) { return ContentService .createTextOutput(JSON.stringify(data)) .setMimeType(ContentService.MimeType.JSON); }
3
Guardar (Ctrl+S) → Implementar → Nueva implementación
• Tipo: Aplicación web
• Ejecutar como: Yo
• Quién tiene acceso: Cualquier usuario ← imprescindible
• Clic en Implementar → si pide autorización → Autorizar acceso → Permitir
4
Copiar URL → Config → pegar en "URL del Web App" → Guardar → Probar conexión
La URL tiene este formato: https://script.google.com/macros/s/[ID]/exec
